\newcommand{\placetextbox}[4] {% \placetextbox{<offset top>}{<offset left/right>}{<align>}{<stuff>}
		\setbox0=\hbox{#4}% Put <stuff> in a box
		\AddToShipoutPictureFG*{% Add <stuff> to current page foreground
				\if#3r
						\put(\LenToUnit{\paperwidth-#1},\LenToUnit{\paperheight-#2}) {
								\vtop{{\null}\makebox[0pt][r] {
												\begin{tabular}{r}
												  #4
												\end{tabular}
										}
								}
						}%
				\else
						\put(\LenToUnit{#1},\LenToUnit{\paperheight-#2}) {
								\vtop{{\null}\makebox[0pt][l] {
												\begin{tabular}{l}
												  #4
												\end{tabular}
										}
								}
						}%
				\fi
		}%
}%

%%% --- definition --- %%%
% not number 
\newenvironment{definition*}[1] {
		\par\noindent
		\begin{minipage}{\textwidth}
				\par\vspace{1em}\textbf{#1:}\vspace{-0.5em}
				\begin{enumerate}\item[]\slshape
				} {
				\end{enumerate}
		\end{minipage}
		\par\noindent
		\vspace{0.5em}
}

% numbered
\newcounter{cudefinitioncounter}
\counterwithin{cudefinitioncounter}{section}
\newenvironment{definition}[1] {
		\par\noindent
		\begin{minipage}{\textwidth}
				\addtocounter{cudefinitioncounter}{1}
				\par\vspace{1em}\textbf{Definition \thecudefinitioncounter} (#1).\vspace{-0.5em}
				\begin{enumerate}\item[]\slshape
				} {
				\end{enumerate}
		\end{minipage}
		\par\noindent
		\vspace{0.5em}
}
%%% --- definition --- %%%

%%% --- problem statement --- %%%

\newenvironment{problemstatement} {
		\subsection*{Problem Statement}
		\vspace{-0.25em}
		\slshape
} {
		\vspace{-1em}
		\subsection*{Solution}
		\vspace{-0.5em}
}
% without solution
\newenvironment{problemstatement*} {
		\subsection*{Problem Statement}
		\vspace{-0.25em}
		\slshape
} {
		\par
}
